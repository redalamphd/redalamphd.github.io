          file_size = os.path.getsize('rimbaud.txt')
          infile = open("rimbaud.txt", 'r+b')
          outfile= open("rimbaud2.txt", 'wb')
          for i in range(file_size) :
              octet = infile.read(1)
              print(octet)
              if octet == b'\x20' : # code hexa ascii de l'espace
                  print("ok")
                  outfile.write(b'\x0D') # code hexa ascii du retour ligne
              else :
                  outfile.write(octet)
          infile.close()
          outfile.close()
